 login(loginForm) {
        this.isLoading = true;
        this.authService.login(loginForm.get('email').value, loginForm.get('password').value)
            .subscribe(data => {
                if (data in RoleTypes) {
                    //console.log(`/${RoleTypes[data]}`);
                        //this.router.navigate([`/${RoleTypes[data]}`]);
                        this.router.navigate(['/admin']);
                } else
                {
                    //this.loginMessage = 'Invalid User Role';
                    this.loginMessage = this.translate.instant("Login.InvalidUserRole")
                }
                }, loginError => {

                    //this.loginMessage = 'Invalid Usersname and/or password';
                    this.loginMessage = this.translate.instant("Login.InvalidUsernameOrPassword")
                    this.isLoading = false;

                }
            );

    }
